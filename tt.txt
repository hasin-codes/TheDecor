        const ctx = gsap.context(() => {
            const slides = slidesRef.current.filter(Boolean);
            const totalSlides = slides.length;

            if (totalSlides < 2 || !trackRef.current || !sectionRef.current) return;

            // ✅ SET EXPLICIT WIDTH
            gsap.set(trackRef.current, {
                width: `${totalSlides * 100}vw`
            });

            // 1. HORIZONTAL SCROLL with VIEWPORT PIN

                x: () => -(trackRef.current!.scrollWidth - window.innerWidth),
                ease: "none",
                scrollTrigger: {
                    trigger: sectionRef.current,
                    start: "top top",
                    end: () => `+=${trackRef.current!.scrollWidth - window.innerWidth}`,
                    scrub: 1,
                    pin: true,
                    pinSpacing: true,        // ✅ ADD

                        ease: "power2.inOut"

                    onUpdate: (self) => {
                        // Update progress bar
                        if (progressRef.current) {
                            gsap.set(progressRef.current, { scaleX: self.progress });
                        }

                        // ✅ FIXED: Velocity Skew Effect - target specific slides
                        const velocity = self.getVelocity();
                        const skew = gsap.utils.clamp(-5, 5, velocity / 300);

                            const content = slide?.querySelector('.testimonial-content');

                                gsap.to(content, {
                                    skewX: skew,
                                    overwrite: 'auto',
                                    duration: 0.1,
                                    ease: "power1.out"
                                });
                            }
                        });
                    }
                }
            });

                        xPercent: 50,
                        ease: "none",
                        scrollTrigger: {
                            trigger: slide,
                            containerAnimation: scrollTween,
                            start: "left right",
                            end: "right left",
                            scrub: true
                        }

            ScrollTrigger.refresh();

            ctx.revert();

                {testimonials.map((item, i) => (
                    <div
                        key={item.id}
                        ref={el => { slidesRef.current[i] = el; }}
                        className="w-screen h-full flex-shrink-0 relative flex flex-col justify-center px-4 py-16 md:px-14 lg:px-24 overflow-hidden border-r border-white/5"
